@startuml

actor Amministratore as ADMIN
participant "AmmPanel" as AP
participant "AmmController" as AC
participant "ProxyAmministratore" as PROXY
participant "AmministratoreService" as SERV
participant "FacadeSingletonDB" as FS
database "MySQL" as DB

== Disattivazione Utenti ==

ADMIN -> AP : Clicca "Disattiva Utente"
activate AP

AP -> AC : disattivaUtente(cf)
activate AC

AC -> PROXY : disattivaUtente(cf)
activate PROXY

PROXY -> PROXY : isAmministratore()
activate PROXY
deactivate PROXY

PROXY -> SERV : disattivaUtente(cf)
activate SERV

SERV -> FS : findPazienteByCf(cf) / findDipendenteByCf(cf)
activate FS

FS -> DB : selectPersonaByCf(cf)
activate DB
DB --> FS : PersonaDB
deactivate DB

FS --> SERV : Persona p
deactivate FS

SERV -> FS : disattivaPersona(p)
activate FS

FS -> DB : query di update stato (disattivazione)
activate DB
DB --> FS : esito (true/false)
deactivate DB

FS --> SERV : utenteDisattivato (boolean)
deactivate FS

SERV --> PROXY : true
deactivate SERV

PROXY --> AC : true
deactivate PROXY

AC -> AP : Mostra PopUp di successo
deactivate AC

AP --> ADMIN : Feedback visivo
deactivate AP

@enduml
