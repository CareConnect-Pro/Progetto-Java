@startuml

actor Utente as U
participant "LoginPanel" as LP
participant "UtentePanel" as UP
participant "FacadeController" as FC
participant "ProxyLogin" as PROXY
participant "LoginService" as SERV
participant "GestoreSessione" as GS
participant "FacadeSingletonDB" as FS
database "MySQL" as DB

== Fase di Login ==

U -> LP : Inserisce cf e pw\nClicca Login
activate LP

LP -> FC : effettuaLogin(cf, pw)
activate FC

FC -> PROXY : login(cf, pw)
activate PROXY

PROXY -> SERV : login(cf, pw)
activate SERV

SERV -> FS : findDipendenteAttivoByCf(cf)
activate FS

FS -> DB : query select dipendente
activate DB
DB --> FS : record Dipendente / null
deactivate DB

FS --> SERV : Dipendente d / null
deactivate FS

alt Credenziali Corrette
    SERV -> GS : setUtenteLoggato(d)
    activate GS
    deactivate GS
end

SERV --> PROXY : d (Dipendente / null)
deactivate SERV

PROXY --> FC : d (Dipendente / null)
deactivate PROXY

alt Login Fallito (d == null)
    FC -> LP : Mostra popup di errore
else Login Eseguito con Successo
    FC -> LP : setVisible(false)
    FC -> UP : setVisible(true)
    activate UP
    deactivate UP
end

deactivate FC
deactivate LP

@enduml
